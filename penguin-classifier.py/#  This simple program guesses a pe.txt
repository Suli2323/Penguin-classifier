#  This simple program guesses a penguin's species using bill length  and bill depth. It imitates what machine learning models do using basic Python rules.

penguins = [
    {"species": "Adelie", "bill_length": 38, "bill_depth": 19},
    {"species": "Chinstrap", "bill_length": 48, "bill_depth": 18},
    {"species": "Gentoo", "bill_length": 50, "bill_depth": 15}
]

# User input
user_length = float(input("Enter bill length (mm): "))
user_depth = float(input("Enter bill depth (mm): "))

# Simple "model"
if user_length < 42:
    guess = "Adelie"
elif user_depth > 17 and user_length > 45:
    guess = "Chinstrap"
else:
    guess = "Gentoo"

print("Predicted species:", guess)
